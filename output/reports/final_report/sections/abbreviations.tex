\section*{Abbreviations and Symbols}

\subsection{Abbreviations}
\begin{itemize}
  \item IIFO -- Intentional Ingestion of Foreign Object(s)
  \item REML -- Restricted Maximum Likelihood
  \item HK -- Hartung–Knapp adjustment
  \item CI -- Confidence Interval
  \item OR -- Odds Ratio
  \item aOR -- Adjusted Odds Ratio
  \item B.B. -- Button Battery
  \item CENTRAL - Cochrane Central Register of Controlled Trials
\end{itemize}

\subsection{Symbols}
\begin{itemize}
  \item $\chi^2$ -- Chi-squared statistic (Cochran’s Q)
  \item $\beta$ -- Regression coefficient
  \item $\tau^2$ -- Between-study variance
\end{itemize}
